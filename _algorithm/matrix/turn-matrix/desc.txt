N X N 배열을
1. 90도회전
2. 정방향뒤집어라
3. 정역방향뒤집어라

arr = [
	[1, 2, 3],
	[4, 5, 6],
	[7, 8, 9]
]

1) 90도회전 = [
	[7, 4, 1],
	[8, 5, 2],
	[9, 6, 3]
]
:: y, x > x, m-y

2) 대각선 = [
	[3, 2, 1],
	[6, 5, 4],
	[9, 8, 7]
]
:: Array.reverse(), 배열을 그냥 역정렬

3) 역대각선 = [
	[1, 4, 7],
	[2, 5, 8],
	[3, 6, 9]
]
:: y, x > m - y, m - x
